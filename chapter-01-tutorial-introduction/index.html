Exercise 1-5 — Primes below N

function isPrime(n) {
  if (n < 2) return false;
  for (let i = 2; i * i <= n; i++)
    if (n % i === 0) return false;
  return true;
}

Exercise 1-4 — Compound Interest Table

function compoundInterest(P, r, n, t) {
  return P * Math.pow(1 + r / n, n * t);
}
